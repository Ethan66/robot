    # compile

    # 导读

    白话版 (opens new window) // 为主 parese (opens new window) // 看总结的上面一段 optimize (opens new window) // 好累，看：逐行剖析 Vue.js 源码 generate (opens new window) // 好累，看：逐行剖析 Vue.js 源码

    若options.render存在，则没有compile这步，vue直接执行options.render函数 compile的作用是把template模板解析生产成render函数，render函数执行后就生成了对应的Vnode。 compile的三步骤：Parse（将template模板生成ast节点(抽象语法树)），Optimize（遍历ast节点，标记为静态的节点，生成新的ast节点(抽象语法树)），Generate（将新的ast节点生成render字符串）。 render字符串转为render函数：render = new Function(render)，将render函数保存在vm.$options.render上。

    # 什么时候执行

    在created和beforeCreate之间，判断$options.render是否存在，不存在，获取template字符串，调用compileToFunctions(template)，compileToFunctions有compile函数，生成render函数。compile函数包含了parse函数, optimize函数, generate函数。

    # Parse

    接收template原始模板，按照模板的节点和数据生成对应的ast（抽象语法树）

    # Optimize

    遍历递归每一个ast节点，找出所有的静态节点打标，找出所有的静态根节点打标。标记静态的节点（没有绑定任何动态数据staticRoot: true）。这样就知道那部分不会变化，于是在页面需要更新时，直接clone就可以了，减少去比对这部分DOM。从而达到性能优化的目的。

    • 怎么判断是否是静态节点：
      1. 从ast根节点开始遍历，false为动态，说明不是静态节点
      1. node.type === 2，动态的文本节点，返回false
      1. node.type === 3，纯文本，返回true
      1. node.type === 1，元素节点，进一步判断
      1. 存在v-pre返回true；否则不存在特殊指令v-if,v-else，不存在slot和component，不存在特殊组件，节点的父节点不能带v-for，template，才能返回true
      1. 节点的返回的true，但是节点的子节点返回false，那么此节点要改回false
    • 找出所有的静态节点，再找出静态根节点：
      1. 从ast根节点开始遍历：节点为静态节点，且节点存在子节点()，且节点不能只有一个文本节点，才能是静态根节点
    ast执行了optimize后的结果。template：<div><span>1</span>{{a}}</div>
    // optimize
{    
  tag: "div",    
  children:[{
    staticRoot: true, // optimize后的静态根节点标识
    tag: "span",        
    children: [{
      static: true, // optimize后的静态文本的标识
      text: '1'
    }]   
  }, {
    static: false, // 表示不是静态节点
    text: '\{\{a\}\}' // 表示双括号a
  }]
}

    # Generate

    把前两步生成完善的 ast 组装成 render 字符串（后面通过createFunction(render)转成函数），保存到vm.$options.render

    template执行了Parse后的结果。template：<div><span>{{b}}</span>{{a}}</div>
    // 前两步后的ast
    {    
      static: false,    
      staticRoot: false,    
      tag: "div",    
      children: [{        
        static: false,        
        staticRoot: false,        
        tag: "span",        
        children: [{            
          static: false,            
          text: "\{\{b\}\}" // 实际为双大括号a，加\是为了显示出来
        }]
      },{        
          static: false,        
          text: "\{\{a\}\}" // 实际为双大括号a，加\是为了显示出来
      }]
    }

    // 经过generate的流程
    // 1、处理最外层的div, code = `_c('div', [`
    // 2、遍历div子节点，遇到span, code = `_c('div', [ _c('span', [`
    // 3、遇到span的子节点b, code = `_c('div', [ _c('span', [ _v(b)`
    // 4、span子节点处理完成，闭合span的子节点数组, code = `_c('div', [ _c('span', [ _v(b) ]`
    // 5、处理span同级子节点, code = `_c('div', [ _c('span', [ _v(b) ], _v(a)`
    // 6、所有子节点处理完成，闭合div的子节点数组, code = `_c('div', [ _c('span', [ _v(b) ], _v(a) )]`

    # 源码

    # Parse

    • template: <div>11</div>
      1. while遍历template，匹配到<div>，执行start方法，将div转为ast节点element，当前没有root节点，将root = element，（若有currentParent，currentParent.children.push(element)），设置currentParent = element，将element保存进节点缓存数组stack，stack === [ {tag:'div' , children:[ ] } ]
      1. 第一轮结束。template === 11</div>
      1. 开头不是<，执行char方法，执行currentParent.children.push({ type:3 , text:'11' })，stack === [ { tag:'div' , children:[ { type:3 , text:'11' } ] } ]
      1. 第二轮结束。template: </div>
      1. 匹配到是结束标签，执行end方法，删除stack的最后一项，currentParent为stack的倒数第二项
      1. 遍历结束，返回root。
    parse(template)，针对一般template复现，不考虑特殊情况（没有结束标签，文本存在<，注释等）
    // 开始解析template
function parse(template) {
  var stack = [] // 闭包：缓存每个头标签，最后一个头标签表示当前标签的父节点
  var root // 闭包：根节点，最终的ast节点保存在这里
  var currentParent // 闭包：当前标签的父节点
  parseHTML(template, {
    start, // 开始标签处理函数
    end, // 结束标签处理函数
    chars // 文本处理函数
  })
  return root
}

function parseHTML(html, options) {
  while(html) {
    var textEnd = html.indexOf('<'),
        text, rest, next
    // html以<开头
    if (textEnd === 0) {
      var endTagMatch = html.match(/^<\/[\w-]+?>/) // 是尾标签
      if (endTagMatch) {
        html = html.substring(endTagMatch[0].length) // html截掉尾标签
        options.end() // stack最后一项去掉，currentParent为stack的最后一项
        continue
      }
    }
    var startTagMatch = html.match(/^<[\w-]+?>/)
    if (startTagMatch) { // 是头标签
      startTagMatch = parseStartTag() // 没写：将头标签根据正则取出每一个属性，转为{tagName: 'div', attrs: [{ name: 1 }]}
      optioins.start(startTagMatch) // 转为ast节点，将ast节点保存到currentParent.children中
      continue
    }
    if (textEnd > 0) { // 是文本
      text = html.substring(0, textEnd)
      html = html.substring(text.length)
    }
    if (options.chars && text) {
      options.chars(text) // 将文本保存到currentParent.children中
    }
  }
}

// 辅助函数: 处理头标签
function start(tag, attrs, unary) {
  var element = createASTElement(tag, attrs, currentParent) // 创建ast节点
  // 省略了v-for, v-if等指令
  if (!root) root = element
  if (currentParent) {
    // 把节点添加到父节点的children中
    currentParent.children.push(element)
    element.parent = currentParent
  }
  if (!unary) { // 不是单标签才存入stack
    currentParent = element
    stack.push(element)
  }
}

// 辅助函数
function createASTElement(tag, attrs, parent) {
  return {        
    type: 1,        
    tag: tag,        
    attrsList: attrs,        
    // 把attrs数组转成对象
    attrsMap: makeAttrsMap(attrs),        
    parent: parent,        
    children: []
  }
}

// 辅助函数：处理尾标签
function end() {    
  // 标签解析结束，移除该标签
  stack.length -= 1
  currentParent = stack[stack.length - 1]
}

// 辅助函数：处理文本
function chars(text) {    
  // 必须存在根节点，不可能用文字开头
  if(!currentParent) return
  var children = currentParent.children   
  // 通过 parseText 解析成字符串，判断是否含有双括号表达式，比如 \{\{a\}\}
  // 如果是有表达式，会存放多一些信息，
  var res = parseText(text)    
  if(res) {
    children.push({            
      type: 2,            
      expression: res.expression, // "我叫"+_s(name)+"，我今年"+_s(age)+"岁了"         
      tokens: res.tokens, // ["我叫", {'@binding': name }, "，我今年", {'@binding': age }, "岁了"]           
      text: text // 我叫\{\{name\}\}，我今年\{\{age\}\}岁了"
    })
  }    
  // 普通字符串，直接存为 字符串子节点
  else if(
    !children.length ||
    children[children.length - 1].text !== ' '
  ) {
    children.push({            
      type: 3,            
      text: text
    })
  }
}
