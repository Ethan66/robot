    # mixins

    # 什么时候合并

    白话版 (opens new window) // 面试的话没问题 源码版 (opens new window) // 不需要断点，这里很清晰

    vue实例化之前，全局设置了Vue.options的三个属性，components（全局组件），directives（全局指令），filters（全局过滤器）。vue实例化时候，在beforeCreate之前，调用mergeOptions(resolveConstructorOptions(vm.constructor), options || {}, vm)方法，对配置的options进行合并。

    # 怎么合并

    # mergeOptions()

    • mergeOptions(resolveConstructorOptions(vm.constructor), options || {}, vm)
    • 参数：resolveConstructorOptions(vm.constructor): 解析constructor上的options属性，一般为{}，当做parent
    • 参数：options: 配置options
    • 校验options规范：校验子组件components的name规范，props是否是对象数组等
    • 合并mixins: 执行mergeOptions(parent, child.mixins[i], vm)，返回值传给parent
    • 先遍历parent的属性key，执行mergeField(key)
    • 后遍历options的属性key，将parent不存在的属性key进行mergeField(key)

    # mergeField(key)

    • strats变量：策略者模式，每个key存着相对应的方法，比如strats.el()，strats.data()。取出来赋值给strat变量
    • 执行方法strat()，将parent的key属性的value和options的key属性的value进行合并，返回的值赋值给新的options的属性key
    • 最后return options，跳出mergeOptions()方法

    # data合并

    data执行顺序：1、options只有data属性，返回值为option.data.call(vm, vm)。2、options有data和mixins属性，函数执行顺序为data(), mixins数组是倒序执行data()。权重是：data()最大，mixins数组后面的权重比前面大（和函数执行顺序相反，因为函数先执行的后合并对象）。详情见data文档。

    strats.data()：childVal为options.data, parentVal为parent.data
    function () {
  return function mergedInstanceDataFn() {
    // instance merge
    var instanceData = typeof childVal === 'function'
        ? childVal.call(vm, vm)
        : childVal;

    var defaultData = typeof parentVal === 'function'
        ? parentVal.call(vm, vm)
        : parentVal;

    if (instanceData) {
        return mergeData(instanceData, defaultData)
    } else {
        return defaultData
    }
  }
}

    # 生命钩子合并

    生命钩子保存到数组，mergeOptions方法先将mixins的生命钩子保存到parentVal中，再合并配置项的生命钩子。所以执行顺序是先执行全局mixins的钩子，再执行组件mixins的钩子，最后执行当前组件的钩子。

    strats.created()：childVal为options.created, parentVal为parent.created
    function mergeHook(parentVal, childVal) {
  return childVal ? 
    (parentVal ? parentVal.concat(childVal) : (Array.isArray(childVal) ? childVal : [childVal]))
      : parentVal
}

    # watch合并

    watch也是保存到数组中，因为watch是对象，所以watch是将mixins和options中相同key对应的属性值保存到一个数组中，所以合并后得到的数据结构是：watch: { a: [function mixins(){}, function options(){}] }，执行顺序是全局-mixins-组件

    strats.watch()：childVal为options.watch, parentVal为parent.watch
    strats.watch = function(parentVal, childVal, vm, key) { 
  if (!parentVal)  return childVal
  var ret = {}
  // 复制 parentVal 到 ret 中
  for (var key in parentVal) {
    ret[key] = parentVal[key];
  }    
  for (var key$1 in childVal) {        
    var parent = ret[key$1];        
    var child = childVal[key$1];        
    if (!Array.isArray(parent)) {
        parent = [parent];
    }
    ret[key$1] = parent ? parent.concat(child) : 
        ( Array.isArray(child) ? child: [child] );
  }    
  return ret
}

    # components, directives, filter合并

    并没有进行遍历合并，而是存在对象的原型上。obj.self_filter: 当前组件的filter，obj.ptoto.mixins_filter: 组件mixins的filter，obj.proto.proto.global_filter: 全局filter。比如components先判断组件的components有没有对应key，没有就直接找mixins中components，一直往原型上找。

    strats.filters()：childVal为options.filters, parentVal为parent.filters
    strats.filters = function mergeAssets(parentVal, childVal, vm, key) {    
  var res = Object.create(parentVal || null)   
  if (childVal) { 
    for (var key in childVal) {
      res[key] = childVal[key]
    }   
  } 
  return res
}

    # props, computed, methods

    唯一性，是不允许重名的。权重谁大用谁的，当前组件>mixins>全局

    strats.computed()：childVal为options.computed, parentVal为parent.computed
    strats.computed = function(parentVal, childVal, vm, key) {    
  if (!parentVal) return childVal
  var ret = Object.create(null);   
  // 把 parentVal 的字段 复制到 ret 中
  for (var key in parentVal) {
    ret[key] = parentVal[key];
  }    
  if (childVal) {        
      for (var key in childVal) {
        ret[key] = childVal[key];
      }
  }    
  return ret
}
