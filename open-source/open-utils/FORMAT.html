    # FORMAT

    # 时间戳转化

    formatDate(now: number | string, fmt?: string): string

    Params:

    now - 时间戳 || 时间字符串

    fmt - 生成的时间格式，默认'yyyy-MM-dd'

    Returns: string

    Example:

    const date = utils.formatDate(123323423423)

    const formatDate = (now: string | number, fmt = 'yyyy-MM-dd'): string => {
  if (typeof now === 'string') {
    now = now.replace(/-/g, '/')
  }
  const date = new Date(now)
  const o = {
    'M+': date.getMonth() + 1, // 月份
    'd+': date.getDate(), // 日
    'h+': date.getHours(), // 小时
    'm+': date.getMinutes(), // 分
    's+': date.getSeconds(), // 秒
    'q+': Math.floor((date.getMonth() + 3) / 3), // 季度
    S: date.getMilliseconds(), // 毫秒
  }
  if (/(y+)/.test(fmt)) {
    fmt = fmt.replace(RegExp.$1, (date.getFullYear() + '').substr(4 - RegExp.$1.length))
  }
  for (const k in o) {
    if (new RegExp('(' + k + ')').test(fmt)) {
      fmt = fmt.replace(RegExp.$1, String(o[k]).padStart(RegExp.$1.length, '0'))
    }
  }
  return fmt
}
