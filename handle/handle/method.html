    # 数组去重

    在原型上自定义方法：[1,2,3,3,4,4].distinct(); //[1,2,3,4]
    // [...new Set(arr)]
// Array.from(new Set(arr))
Array.prototype.distinct = function() {
  const map = {}
  const result = []
  for (const n of this) {
      if (!(n in map)) {
          map[n] = 1
          result.push(n)
      }
  }
  return result
}
[1,2,3,3,4,4].distinct(); //[1,2,3,4]

    Array.prototype.bubleSort=function () {
  let arr=this,
      len = arr.length;
  for (let outer = len; outer >= 2; outer--) {
    for (let inner = 0; inner <= outer - 1; inner++) {
      if (arr[inner] > arr[inner + 1]) {
        //升序
        [arr[inner], arr[inner + 1]] = [arr[inner + 1], arr[inner]];
        console.log([arr[inner], arr[inner + 1]]);
      }
    }
  }
  return arr;
}

    # 对象监听

    监听对象里的所有属性，属性改变了console一下
    let info = {};
function observe(obj) {
  if (!obj || typeof obj !== "object") {
    return;
  }
  for (var i in obj) {
    definePro(obj, i, obj[i]);
  }
}

function definePro(obj, key, value) {
  observe(value);
  Object.defineProperty(obj, key, {
    get: function() {
      return value;
    },
    set: function(newval) {
      console.log("检测变化", newval);
      value = newval;
    }
  });
}
definePro(info, "friends", { name: "张三" });
info.friends.name = "李四";

    type = 'orientationchange' in window ? 'orientationchange' : 'resize',

var imgList = document.querySelectorAll('img')
// 可视窗口高度
var clientHeight = window.innerHeight || document.documentElement.offsetHeight || documnent.body.offsetHeight
window.addEventListener('scroll', function () {
  var scrollTop = document.documentElement.scrollTop || document.body.scrollTop;
  [...imgList].forEach(img => {
    var diff = clientHeight + scrollTop - img.offsetTop
    if (diff > 0 && diff < img.offsetHeight + clientHeight) {
      img.src = img.getAttribute('data')
    }
