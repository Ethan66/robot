    # zod：运行时ts

    # 使用场景

    1. typescript只能进行编译时检查，运行时转为js，并没有校验类型的功能，所以需要使用运行时的类型校验。
    2. 测试环境：有些接口返回的字段太多，影响bug排查速度，或者字段值太过隐晦，比如约定返回'1','2'，结果返回了1, 2，所以知道了返回了非约定的字段值，直接把bug指派给后端。
    3. 线上环境：和后端约定了对应的字段值，防止后端在后续版本中返回了不符合约定的字段值，导致前端报错，让前端一度以为自己没有做好兜底，比如没有考虑字段值为null的情况。

    # 基本类型

    import { z } from 'zod';

const objSchema = z.object({
  name: z.string(),
  name2: z.enum(['a', 'b', 'c']), // 'a' | 'b' | 'c'
  age: z.number(),
  age2: z.union([z.literal(0), z.literal(1)]), // 1 | 2
  isStudent: z.boolean(),
  birthday: z.date(),
  null: z.null(),
  arr: z.array(z.string()),
}).passthrough()

    # 实战

    # 接口响应数据校验公共方法

    import { z } from 'zod'

export const validateData = <T>(schema: z.ZodType<T>, data: unknown): T | null => {
  try {
    return schema.parse(data);
  } catch (error) {
    if (error instanceof z.ZodError) {
      const errs = error.errors.reduce((errResult, cur) => {
        const val = cur.path.reduce((objVal: any, key) => objVal[key], data)
        const path = cur.path.join('.')
        errResult[path] = `val: ${val}, errMsg: ${cur.message}`
        return errResult
      }, {} as any)
      console.error('zod validate error: ', errs)
    } else {
      console.error('zod unexpected error:', error);
    }
    return data as any;
  }
};
