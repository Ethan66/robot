    # 遍历: _.map => []

    • 数组：等于原生的map，还可以拿到对象的某个属性，比如labelName；
    • 对象：对对象进行遍历，可以获取values的集合，相当于对象的map函数
    // 数组
// 获取name属性的值的集合
let a = _.map(arr, 'name'); // result: ['开启', '关闭']

// 对象
// 获取age属性的值的集合
let b = _.map(obj, (val, key, obj) => {return val.age * 2 }); // result: [72, 80, 2]

    # 筛选: _.filter => []

    • 数组：对对象数组中多个属性进行筛选
    • 对象：对对象对象中多个属性进行筛选，得到的values，也可传一个函数
    // 数组
// 获取name属性的值的集合
let a = _.filter(arr, { name: '开启' }); // result: [{ name: '开启', id: 1 }]
// 获取name和error属性的值的集合
a = _.filter(arr, { name: '开启', error: 1 }); // result: []

// 对象
// 获取age属性的值的集合
let b = _.filter(obj, { active: false }); // result: [{ age: 40, active: false }]

    # 找到: _.find

    • 数组：对对象数组中多个属性进行筛选，得到一个对象
    • 对象：对对象对象中多个属性进行筛选，得到一个value
    // 数组
// 获取name属性的值的集合
let a = _.find(arr, { name: '开启' }); // result: { name: '开启', id: 1 }
// 获取id属性的值的集合
a = _.find(arr, { name: '开启', error: 1 }); // result: undefined

// 对象
// 获取age属性的值的集合
let b = _.find(obj, { active: false }); // result: { age: 40, active: false }
