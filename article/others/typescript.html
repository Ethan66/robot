    # typescript常用函数记录

    # Pick<T, K>

    作用：Pick 从类型 T 中选择一部分键 K，并创建一个新的类型。

    interface User {
  id: number; name: string; email: string;
}

type UserContactInfo = Pick<User, 'name' | 'email'>; // 从 User 类型中挑选 name 和 email 属性

const user: UserContactInfo = {
  name: 'John Doe',
  email: 'john@example.com',
};

// 源码
/* type Pick<T, K extends keyof T> = {
  [P in K]: T[P];
}; */

    # Omit<T, K>

    作用：Omit 从类型 T 中排除指定的键 K，并创建一个新的类型。

    interface User {
  id: number; name: string; email: string;
}

type UserWithoutId = Omit<User, 'id'>; // 从 User 类型中排除 id 属性

const user: UserWithoutId = {
  name: 'John Doe',
  email: 'john@example.com',
};

// 源码
/* type Omit<T, K extends keyof any> = Pick<T, Exclude<keyof T, K>>; */

    # Partial< T >

    作用：Partial 将类型 T 的所有属性变为可选的。

    interface User {
    id: number; name: string; email: string;
}

type PartialUser = Partial<User>; // 将 User 类型中的所有属性变为可选的

const user: PartialUser = {
  name: 'John Doe',
  email: 'john@example.com',
};

// 源码
/* type Partial<T> = {
  [P in keyof T]?: T[P];
}; */

    # Readonly< T >

    作用：Readonly 将类型 T 的所有属性变为只读的。

    interface User { id: number; }

type ReadonlyUser = Readonly<User>; // 将 User 类型中的所有属性变为只读的

const user: ReadonlyUser = { id: 1 };

user.id = 2; // Error: Cannot assign to 'id' because it is a read-only property.

    # Required< T >

    作用：Required 将类型 T 的所有属性变为必选的。

    interface User {
  id?: number; name?: string; email?: string;
}

type CompleteUser = Required<User>; // 将 User 类型中的所有属性变为必选的

const user: CompleteUser = {
  id: 1,
  name: 'Alice',
  email: 'alice@example.com',

    # Exclude<T, U>

    作用：Exclude 从类型 T 中排除 U 中存在的类型。

    type T = 'a' | 'b' | 'c';
type U = 'a';

// 从 T 中排除 U 中存在的类型
type Result = Exclude<T, U>; // 'b' | 'c'

    # Extract<T, U>

    作用：Extract 从类型 T 中提取 U 中存在的类型。

    type T = 'a' | 'b' | 'c';
type U = 'a' | 'd';

// 从 T 中提取 U 中存在的类型
type Result = Extract<T, U>; // 'a'

    # NonNullable< T >

    作用：NonNullable 从类型 T 中排除 null 和 undefined。

    type T = string | number | null | undefined;

// 从 T 中排除 null 和 undefined
type Result = NonNullable<T>; // string | number
